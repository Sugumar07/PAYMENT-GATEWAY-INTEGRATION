#include<stdio.h>
int main()
{
    int n,a[100];
    scanf("%d",&n);
    for(int i=0;i<n;i++)
    {
        scanf("%d",&a[i]);
    }
    int sum=0;
    sum=a[0];
    for(int i=0;i<n;i++)
    {
        if(a[i]%2==1 && a[i-1]%2==0)
        {
            continue;
        }
        else
        {
            sum=sum+a[i];
        }
    }
    printf("%d",sum);
}
